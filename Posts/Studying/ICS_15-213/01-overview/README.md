## Demo

```c
#include <stdio.h>
#include <stdlib.h>

int sq(int x) {
    return x*x;
}

// argc：命令行参数个数；argv[]：命令行参数数组
int main(int argc, char *argv[]) {
    int i;
    for (i = 1; i < argc; i++) {
	int x = atoi(argv[i]);
	int sx = sq(x);
	printf("sq(%d) = %d\n", x, sx);
    }
    return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int a[2];
    double d;
} struct_t;

double fun(int i) {
    volatile struct_t s;
    s.d = 3.14;
    s.a[i] = 1073741824; /* Possibly out of bounds */
    return s.d; /* Should be 3.14 */
}

int main(int argc, char *argv[]) {
    int i = 0;
    if (argc >= 2)
	i = atoi(argv[1]);
    double d = fun(i);
    printf("fun(%d) --> %.10f\n", i, d);
    return 0;
}
```

- `atoi`
  - 转换字符串为整型
  - `int atoi (const char * str);`
- `volatile`
  - 告知编译器不做任何优化
  - 每次都从内存读取修饰的变量（不重复使用寄存器中的缓存）
